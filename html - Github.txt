    function setLang(lang) {
      const t = translations[lang];
      if (!t) return;

      document.getElementById("title-text").textContent = t.title;
      document.getElementById("subtitle-text").textContent = t.subtitle;
      document.getElementById("tagline-text").textContent = t.tagline;
      document.getElementById("btn-details").textContent = t.btnDetails;
      document.getElementById("btn-whitepaper").textContent = t.btnWhitepaper;

      document.getElementById("about-title").textContent = t.aboutTitle;
      document.getElementById("about-text").textContent = t.aboutText;

      document.getElementById("token-title").textContent = t.tokenTitle;
      document.getElementById("spec-ticker-label").textContent = t.specTicker;
      document.getElementById("spec-supply-label").textContent = t.specSupply;
      document.getElementById("spec-dist-label").textContent = t.specDist;
      document.getElementById("spec-nominal-label").textContent = t.specNominal;
      document.getElementById("spec-staking-label").textContent = t.specStaking;
      document.getElementById("spec-staking-text").textContent = t.specStakingText;
      document.getElementById("spec-usecase-label").textContent = t.specUsecase;
      document.getElementById("spec-usecase-text").textContent = t.specUsecaseText;

      document.getElementById("utility-title").textContent = t.utilityTitle;
      document.getElementById("utility-intro").textContent = t.utilityIntro;

      const listEl = document.getElementById("utility-list");
      listEl.innerHTML = "";
      t.utilityList.forEach(item => {
        const li = document.createElement("li");
        li.textContent = item;
        listEl.appendChild(li);
      });

      document.getElementById("utility-first-terminal").textContent = t.firstTerminal;
      document.getElementById("utility-next-steps").textContent = t.nextSteps;

      document.getElementById("wp-title").textContent = t.wpTitle;
      document.getElementById("wp-text-1").textContent = t.wpText1;
      document.getElementById("wp-text-2").textContent = t.wpText2;
      document.getElementById("btn-wp-pdf").textContent = t.wpPdf;
      document.getElementById("btn-wp-docx").textContent = t.wpDocx;

      // badges
      document.querySelectorAll("[data-i18n-badge]").forEach(el => {
        const key = el.getAttribute("data-i18n-badge");
        if (t.badges[key]) el.textContent = t.badges[key];
      });

      document.querySelectorAll(".lang-btn").forEach(btn => {
        btn.classList.toggle("active", btn.getAttribute("data-lang") === lang);
      });
    }

    document.querySelectorAll(".lang-btn").forEach(btn => {
      btn.addEventListener("click", () => {
        const lang = btn.getAttribute("data-lang");
        setLang(lang);
      });
    });

    // default EN
    setLang("en");
  </script>
